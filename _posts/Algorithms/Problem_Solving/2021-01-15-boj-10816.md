```cpp
#include<iostream>
#include<vector>
#include<unordered_map>

using namespace std;

void solution(){
    int n; cin>>n;

    unordered_map<int, int> cards;

    for(int idx=0; idx<n; idx++){
        int num; cin>>num;
        cards[num]++;
    }

    int m; cin>>m;
    for(int idx=0; idx<m; idx++){
        int num; cin>>num;
        cout<<cards[num]<<" ";
    }

}
int main(){
    ios_base::sync_with_stdio(false);
	cin.tie(NULL);
    
    solution();

    return 0;
}
```
