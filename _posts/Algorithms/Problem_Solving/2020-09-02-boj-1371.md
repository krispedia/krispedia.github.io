```cpp
#include<iostream>

#include<vector>

#include<string>

using namespace std;

int main(){
    ios::sync_with_stdio(false);
    cin.tie(0); cout.tie(0);
    int c[26]={0};
    string s;
    while(cin>>s){
        for(int i=0; i<s.length(); i++){
            int index = int(s[i])-'a';
            c[index]++;
        }
    }
    int max=0;
    vector<int> res;
    for(int i=0; i<26; i++){
        if(c[i]>max) {
            max = c[i];
            res.clear();
            res.push_back(i);
        }
        else if(c[i]==max){
            res.push_back(i);
        }
    }
    for(int i=0; i<res.size(); i++) {
        cout << char('a' + res[i]);
    }

    return 0;
}
```

## 풀이<br>
각 소문자 알파벳의 수를 세면 되므로 크기가 26인 배열에 카운팅만 하면 되는 간단한 문제이다.  

복병은 입력.  

입력이 끝나는 조건이 주어지지 않아서 `cin.eof()`, `cin.peek()` 등등을 했으나   
`while(cin>>s)` 를 사용하니 되었다.
