#include<iostream>
#include<vector>
#include<queue>

using namespace std;

void solution(int n){

    queue<int> router;

    while(1){
        int num; cin>>num;
        if(num==-1) break;
        
        if(num==0) router.pop();
        else if(router.size()>=n) continue;
        else{
            router.push(num);
        }
    }

    if(router.empty()) cout<<"empty";
    else{
        while(!router.empty()){
            cout<<router.front()<<" ";
            router.pop();
        }
    }
}
int main(){
    ios_base::sync_with_stdio(false);
	cin.tie(NULL);

    int n; cin>>n;
    
    solution(n);

    return 0;
}
