## 문제 링크<br>
 [https://www.acmicpc.net/problem/11053](https://www.acmicpc.net/problem/11053)<br>

#### 전체코드<br>
```cpp
#include<iostream>
#define SIZE 1000
using namespace std;

int main(void){
    int arr[SIZE] = {0};
    int dp[SIZE] = {0};
    int n;
    cin>>n;

    for(int i=0; i<n; ++i)
        cin>>arr[i];

    int max = 0;

    for(int i=0; i<n; ++i){
        int prev_max = 0;
        for(int j=0; j<i; ++j){
            if((arr[j]<arr[i])&&(dp[j] > prev_max)) prev_max = dp[j];
        }
        dp[i] = prev_max+1;

        if(dp[i] > max) max = dp[i];
    }

    cout<<max<<endl;

    return 0;
}
```
