#include<iostream>
#include<vector>
#include<stack>

using namespace std;

void solution(vector<int> inputs){
    int answer = 0;

    stack<int> st;

    for(auto& target: inputs){
        if(target == 0) st.pop();
        else st.push(target);
    }

    while(!st.empty()){
        answer += st.top();
        st.pop();
    }

    cout<<answer;
}

int main(){
    ios_base::sync_with_stdio(false);
	cin.tie(NULL);

    int n; cin>>n;
    vector<int> inputs;
    
    for(int idx=0; idx<n; idx++){
        int in; cin>>in;
        inputs.push_back(in);
    }
    
    solution(inputs);

    return 0;
}
