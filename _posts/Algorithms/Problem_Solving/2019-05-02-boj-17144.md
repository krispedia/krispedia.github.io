삼성 SW 문제 중 하나인 17144번.

미세먼지의 확산, 공기청정기의 시계, 반시계방향 작동을 확인해봤을땐 이상 없었고 테스트 데이터를 돌렸을 때 원하는 출력이 나왔지만 제출하니 '틀렸습니다' 라고 나왔다. 

어디가 틀렸다는 건지.

아직 생각해내지 못한 예외가 있는듯! 아직 예외가 생각나지 않아 방황...ing

```cpp
#include<iostream>
#define MAX 50

using namespace std;

int grid[MAX][MAX];
int temp[MAX][MAX];
int dir[4][2] = { {0,1},{1,0},{0,-1},{-1,0} };
int R,C,T;
void readData(){
    ios_base::sync_with_stdio(false);
    cin>>R>>C>>T;

    for(int i=0; i<R; i++){
        for(int j=0; j<C; j++){
            cin>>grid[i][j];
        }
    }
}
void printGrid(){
    for(int i=0; i<R; i++){
        for(int j=0; j<C; j++){
            cout<<grid[i][j]<<" ";
        }
        cout<<'\n';
    }
}
int spreadMount(int x, int y){
    return grid[x][y]/5;
}
bool moveOK(int x, int y){
    if(x<0 || y<0 || x>=R || y>=C || grid[x][y] == -1)
        return false;
    return true;
}
int spreadDir(int x, int y, int dirIndex, int count){
    if(dirIndex == 4)
        return count;
    else{
        int newX = x+dir[dirIndex][0];
        int newY = y+dir[dirIndex][1];
        if(moveOK(newX, newY)){
            temp[newX][newY] += spreadMount(x,y);
            count++;
        }
        return spreadDir(x,y,dirIndex+1, count);
    }
}
void duplicateGridToTemp(){
    for(int i=0; i<R; i++){
        for(int j=0; j<C; j++){
            temp[i][j] = grid[i][j];
        }
    }
}
void combineGridWithTemp(){
    for(int i=0; i<R; i++){
        for(int j=0; j<C; j++){
            grid[i][j] = temp[i][j];
        }
    }
}
void spread(){
    duplicateGridToTemp();
    for(int x=0; x<R; x++){
        for(int y=0; y<C; y++){
            int moveCount = spreadDir(x, y, 0, 0);
            temp[x][y] -= moveCount*spreadMount(x,y);
        }
    }
    combineGridWithTemp();
}
int getCleanerLoc(){
    for(int i=0; i<R; i++){
        if(grid[i][0] == -1)
            return i;
    }
    return -1;
}
void counterClockWise(int row){
    for(int i=row-2; i>=0; i--)
        grid[i+1][0] = grid[i][0];
    for(int j=1; j<C; j++)
        grid[0][j-1] = grid[0][j];
    for(int i=1; i<=row; i++)
        grid[i-1][C-1] = grid[i][C-1];
    for(int j=C-2; j>0; j--)
        grid[row][j+1] = grid[row][j];

    grid[row][1] = 0;
}
void clockWise(int row){
    for(int i= row+2; i<C; i++)
        grid[i-1][0] = grid[i][0];
    for(int j=1; j<C; j++)
        grid[R-1][j-1] = grid[R-1][j];
    for(int i=R-2; i>=row; i--)
        grid[i+1][C-1] = grid[i][C-1];
    for(int j=C-2; j>=1; j--)
        grid[row][j+1] = grid[row][j];

    grid[row][1] = 0;
}
void airCleaner() {
    int loc = getCleanerLoc();
    counterClockWise(loc);
    clockWise(loc+1);
}
int dustCount() {
    int count=0;
    for(int i=0; i<R; i++){
        for(int j=0; j<C; j++){
            if(grid[i][j] == -1)
                continue;
            count += grid[i][j];
        }
    }
    return count;
}
int main(void) {

    readData();
    cout<<'\n';
    cout<<"[-- INPUT --]"<<'\n';
    printGrid();
    cout<<'\n';

    for(int t=0; t<T; t++){

        spread();
        cout<<"[- AFTER SPREAD -]"<<'\n';
        printGrid();
        cout<<'\n';

        cout<<"[- AFTER CLEANING -]"<<'\n';
        airCleaner();
        printGrid();
        cout<<'\n';
    }
    cout<<dustCount()<<'\n';
    return 0;
}
```
