#include<iostream>
#include<vector>
#include<string>
#include<list>

using namespace std;

void solution(int n){
    list<int> lt;
    for(int idx=1; idx<=n; idx++) lt.push_back(idx);

    while(lt.size()>1){
        // 제일 위 버리고 
        lt.pop_front();
        // 다음 장 아래로 
        int temp = lt.front();
        lt.pop_front();
        lt.push_back(temp);
    }

    cout<<lt.front();
}

int main(){
    ios_base::sync_with_stdio(false);
	cin.tie(NULL);

    int n; cin>>n;
    
    solution(n);

    return 0;
}
