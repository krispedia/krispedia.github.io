#### 전체코드<br>
```cpp
#include<iostream>
#define SIZE 91
using namespace std;

int main(void){
    long dp[SIZE] = {0};
    int n;

    cin>>n;

    dp[1] = 1;
    dp[2] = 1;

    for(int i=3; i<=n; ++i)
        dp[i] = dp[i-1]+dp[i-2];

    cout<<dp[n]<<endl;

    return 0;
}
```
